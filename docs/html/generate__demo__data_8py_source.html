104def create_demo_friendships(db: Session, users):
105 """创建示例好友关系。"""
106 # 为每个用户创建 2-4 个好友
107 for user in users:
108 # 随机选择 2-4 个其他用户作为好友
109 potential_friends = [u for u in users if u.id != user.id]
110 num_friends = min(random.randint(2, 4), len(potential_friends))
111 friends = random.sample(potential_friends, num_friends)
112
113 for friend in friends:
114 # 检查是否已经是好友
115 existing = friendship_crud.get_friendship_by_users(db, user.id, friend.id)
116 if not existing:
117 # 创建好友关系，50% 概率为已接受状态
118 status = random.choice([FriendshipStatus.PENDING, FriendshipStatus.ACCEPTED])
119 friendship = friendship_crud.create_friendship(db, user.id, friend.id)
120 if status == FriendshipStatus.ACCEPTED:
121 friendship_crud.update_friendship_status(db, friendship.id, FriendshipStatus.ACCEPTED)
122 print(f"已创建好友关系: {user.username} -> {friend.username} ({status.value})")

125def create_demo_registrations(db: Session, users, conferences):
126 """创建示例会议注册。"""
127 # 为每个会议随机注册 3-6 个用户
128 for conference in conferences:
129 # 排除创建者（因为创建者不需要注册）
130 potential_attendees = [u for u in users if u.id != conference.creator_id]
131 num_attendees = min(random.randint(3, 6), len(potential_attendees))
132 attendees = random.sample(potential_attendees, num_attendees)
133
134 for attendee in attendees:
135 # 检查是否已经注册
136 existing = registration_crud.get_registration_by_user_and_conference(db, user_id=attendee.id, conference_id=conference.id)
137 if not existing:
138 # 创建注册
139 registration = RegistrationCreate(conference_id=conference.id)
140 registration_crud.create_registration(db, registration=registration, user_id=attendee.id)
141 print(f"已创建会议注册: {attendee.username} -> {conference.title}")

144def create_demo_notifications(db: Session, users, conferences):
145 """创建示例通知。"""
146 # 为每个用户创建 3-5 个随机通知
147 notification_types = list(NotificationType)
148
149 for user in users:
150 num_notifications = random.randint(3, 5)
151
152 for _ in range(num_notifications):
153 # 随机选择通知类型
154 notification_type = random.choice(notification_types)
155
156 # 随机选择发送者（可能为空）
157 sender = random.choice([None] + [u for u in users if u.id != user.id])
158 sender_id = sender.id if sender else None
159
160 # 根据通知类型生成内容和相关 ID
161 content = ""
162 related_id = None
163
164 if notification_type == NotificationType.FRIEND_REQUEST:
165 if sender:
166 content = f"{sender.username} 向您发送了好友请求"
167 # 查找或创建一个好友关系作为相关 ID
168 friendship = friendship_crud.get_friendship_by_users(db, sender_id, user.id)
169 if not friendship:
170 friendship = friendship_crud.create_friendship(db, sender_id, user.id)
171 related_id = friendship.id
172
173 elif notification_type == NotificationType.FRIEND_ACCEPTED:
174 if sender:
175 content = f"{sender.username} 接受了您的好友请求"
176 # 查找或创建一个好友关系作为相关 ID
177 friendship = friendship_crud.get_friendship_by_users(db, user.id, sender_id)
178 if not friendship:
179 friendship = friendship_crud.create_friendship(db, user.id, sender_id)
180 friendship_crud.update_friendship_status(db, friendship.id, FriendshipStatus.ACCEPTED)
181 related_id = friendship.id
182
183 elif notification_type == NotificationType.MEETING_INVITATION:
184 if sender and conferences:
185 conference = random.choice(conferences)
186 content = f"{sender.username} 邀请您参加会议 '{conference.title}'"
187 related_id = conference.id
188
189 elif notification_type == NotificationType.MEETING_JOINED:
190 if conferences:
191 conference = random.choice(conferences)
192 content = f"您已成功加入会议 '{conference.title}'"
193 related_id = conference.id
194
195 elif notification_type == NotificationType.MEETING_LEFT:
196 if conferences:
197 conference = random.choice(conferences)
198 content = f"您已退出会议 '{conference.title}'"
199 related_id = conference.id
200
201 # 创建通知
202 if content:
203 # 随机设置已读状态
204 is_read = random.choice([True, False])
205 notification = notification_crud.create_notification(
206 db, user_id=user.id, type=notification_type, content=content,
207 sender_id=sender_id, related_id=related_id
208 )
209 if is_read:
210 notification_crud.mark_notification_as_read(db, notification.id)
211 print(f"已创建通知: {user.username} <- {notification_type.value} ({content[:30]}...)")
