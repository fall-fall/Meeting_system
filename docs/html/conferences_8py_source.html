26@router.get("/", response_class=HTMLResponse)

30):
31 """显示所有会议列表"""

38 from app.dependencies import get_current_user

43 except:

125@router.get("/{conference_id}", response_class=HTMLResponse)
126async def conference_detail(
127 request: Request,
128 conference_id: int,
129 db: Session = Depends(get_db),
130):
131 """Get a conference by ID."""
132 # 获取当前用户（如果已登录）
133 current_user = None
134 unread_notification_count = 0
135 token = request.cookies.get("access_token")
136 if token and token.startswith("Bearer "):
137 try:
138 from app.dependencies import get_current_user
139 current_user = await get_current_user(token.replace("Bearer ", ""), db)
140 # 获取未读通知数量
141 if current_user:
142 unread_notification_count = notification_crud.get_unread_notification_count(db, current_user.id)
143 except:
144 pass
145
146 conference = conference_crud.get_conference(db, conference_id=conference_id)
147 if not conference:
148 raise HTTPException(status_code=404, detail="未找到会议")
149
150 # Get agenda items
151 agenda_items = conference_crud.get_agenda_items(db, conference_id=conference_id)
152
153 # Check if user is registered
154 is_registered = False
155 if current_user:
156 for registration in conference.registrations:
157 if registration.user_id == current_user.id:
158 is_registered = True
159 break
160
161 return templates.TemplateResponse(
162 "conferences/detail.html",
163 {
164 "request": request,
165 "conference": conference,
166 "agenda_items": agenda_items,
167 "current_user": current_user,
168 "is_registered": is_registered,
169 "unread_notification_count": unread_notification_count,
170 },
171 )
