26def get_friendship_by_users(db: Session, user_id: int, friend_id: int):
27 """获取两个用户之间的好友关系
28
29 Args:
30 db: 数据库会话
31 user_id: 用户ID
32 friend_id: 好友ID
33
34 Returns:
35 Friendship: 好友关系对象，如果不存在则返回None
36
37 Note:
38 此函数会查找两个用户之间的好友关系，无论谁是发起者
39 """
40 return (
41 db.query(Friendship)
42 .filter(
43 or_(
44 # 用户是发起者，好友是接收者
45 and_(Friendship.user_id == user_id, Friendship.friend_id == friend_id),
46 # 好友是发起者，用户是接收者
47 and_(Friendship.user_id == friend_id, Friendship.friend_id == user_id),
48 )
49 )
50 .first()
51 )

77def get_friends(db: Session, user_id: int):
78 """获取用户的所有好友
79
80 Args:
81 db: 数据库会话
82 user_id: 用户ID
83
84 Returns:
85 List[User]: 好友用户对象列表
86 """
87 # 获取用户发起的已接受的好友关系
88 sent_friendships = (
89 db.query(Friendship)
90 .filter(
91 Friendship.user_id == user_id, # 用户是发起者
92 Friendship.status == FriendshipStatus.ACCEPTED # 状态为已接受
93 )
94 .all()
95 )
96
97 # 获取用户接收的已接受的好友关系
98 received_friendships = (
99 db.query(Friendship)
100 .filter(
101 Friendship.friend_id == user_id, # 用户是接收者
102 Friendship.status == FriendshipStatus.ACCEPTED # 状态为已接受
103 )
104 .all()
105 )
106
107 # 提取好友ID
108 friend_ids = [f.friend_id for f in sent_friendships] + [f.user_id for f in received_friendships]
109
110 # 获取好友用户对象
111 return db.query(User).filter(User.id.in_(friend_ids)).all()
